This directory contains the general UI modules what doesn't belongs to any existing namespace

Initialize in html:

    <!-- ui/utilities js START -->
    <script src="../core/ui/utilities/ui.js"></script>
    <script src="../core/ui/utilities/badgeText.js"></script>

Initialize in manifest content scripts:

    /* ui/utilities js START */
    "core/ui/utilities/ui.js",
    "core/ui/utilities/badgeText.js",
